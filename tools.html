<script src="https://cdnjs.cloudflare.com/ajax/libs/lottie-web/5.12.2/lottie.min.js"></script>
<div id="menu-btn"></div>

<script>
  const menuBtn = document.getElementById('menu-btn');

  const menuAnimation = lottie.loadAnimation({
    container: menuBtn,
    renderer: 'svg',
    loop: false,
    autoplay: false,
    path: 'menuV4.json' // Make sure this path is correct
  });

  let isOpen = false;

  menuBtn.addEventListener('click', () => {
    if (isOpen) {
      // Play in reverse
      menuAnimation.setDirection(-1);
      menuAnimation.play();
    } else {
      // Play forward
      menuAnimation.setDirection(1);
      menuAnimation.play();
    }
    isOpen = !isOpen;
  });
</script>
